 ### 运行PS1脚本

```
$timestamp = Get-Date -Format "yyyy-MM-dd_HHmm"
$dir="/home/html/"
$name="qc"
$targetDir = "${dir}${name}" 
$backupDir = "${targetDir}_backup_${timestamp}"
$publishDir="D:\AstCode\src\frontEnd\Ast-QC-Web\dist"
 

$cmd=@"
 mv $targetDir $backupDir
"@

 ssh -o StrictHostKeyChecking=no 'root@8.219.218.2' $cmd

 scp -r $publishDir root@8.219.218.2:$dir

 $cmd = "cd $dir && mv dist $name"
 ssh -o StrictHostKeyChecking=no 'root@8.219.218.2' $cmd
```
